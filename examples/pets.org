** Queries
*** select query for OP domain/range
#+begin_src sparql :url "my-ont.omn" :eval never-export :exports both
  select *
  {
    ?OP a owl:ObjectProperty .
    optional { ?OP rdfs:domain ?domain }
    optional { ?OP rdfs:range ?range }
  } 
#+end_src

*** construct query ~my-construct~ for class taxonomy and OP domain/range
#+name: my-construct
#+begin_src sparql :url "my-ont.omn" :eval never-export :format ttl :wrap "src ttl" :cache yes :post kill-prefixes(data=*this*)
  construct {
    ?class a owl:Class .
    ?subclass rdfs:subClassOf ?class .
    ?OP a owl:ObjectProperty ; rdfs:domain ?domain ; rdfs:range ?range .
  } {
    ?class a owl:Class .
    optional { ?subclass rdfs:subClassOf ?class }
    optional { ?OP rdfs:domain ?domain }
    optional { ?OP rdfs:range ?range }
  } 
#+end_src

*** construct query for axiom annotations
#+name: my-axiom-annotations
#+begin_src sparql :url "my-ont.omn" :eval never-export :format ttl :wrap "src ttl" :cache no :post kill-prefixes(data=*this*) 
  construct {
    ?class a owl:Class .
    ?OP a owl:ObjectProperty ; ?rdfs_prop ?class .
    ?axiom a owl:Axiom ; owl:annotatedSource ?OP ; ?ann_prop ?ann_target .
  } {
    values ?rdfs_prop { rdfs:domain rdfs:range }
    ?class a owl:Class .
    ?OP ?rdfs_prop ?class .
    optional { 
      ?axiom a owl:Axiom ;
           owl:annotatedSource ?OP ;
           ?ann_prop ?ann_target .
    } 
  } 
#+end_src

#+RESULTS[ed9dfcf72060ba3d0de92bf7f5c992d6f6e937f8]: my-axiom-annotations
#+begin_src ttl

ex:cat  a       owl:Class .

[ a                      owl:Axiom ;
  rdfs:seeAlso           <https://www.wikidata.org/wiki/Q1839152> ;
  owl:annotatedProperty  rdfs:range ;
  owl:annotatedSource    ex:chases ;
  owl:annotatedTarget    ex:cat ;
  skos:example           "Tom Cat is chased by Spike, AKA Butch or Killer."
] .

ex:dog  a       owl:Class .

[ a                      owl:Axiom ;
  rdfs:comment           "Certain dogs have found this axiom offensive. Others, not." , "Consult popular cartoons for excellent evidence of this domain axiom." ;
  owl:annotatedProperty  rdfs:domain ;
  owl:annotatedSource    ex:chases ;
  owl:annotatedTarget    ex:dog
] .

ex:chases  a         owl:ObjectProperty ;
        rdfs:domain  ex:dog ;
        rdfs:range   ex:cat .
#+end_src
